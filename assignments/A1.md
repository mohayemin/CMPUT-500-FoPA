## 1.e.
```ql
import java

predicate isInterfaceMethodImpl(Method m) {
	exists (Interface i, Method m1 | 
      m1.getDeclaringType() = i and
      m.overridesOrInstantiates(m1)
  )
}

from Method m
where
	not isInterfaceMethodImpl(m)
select m
```

      m1.getDeclaringType() = i and
      m.overridesOrInstantiates(m1)
  )
}
